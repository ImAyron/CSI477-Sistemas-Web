// Creating tables
Table items as I {
  id int [pk, increment] // auto-increment
  descricao varchar
  created_at timestamp
  updated_at timestamp
}

Table entidades as E {
  id int [pk, increment] // auto-increment
  nome varchar
  bairro varchar
  cidade varchar
  estado varchar
  created_at timestamp
  updated_at timestamp
 }

Table coletas as C {
  id int [pk, increment] // auto-increment
  item_id int [ref: > I.id]
  entidade_id int [ref: > E.id]
  quantidade decimal
  data date
  created_at timestamp
  updated_at timestamp
 }
